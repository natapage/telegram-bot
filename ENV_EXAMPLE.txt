=============================================================================
ПРИМЕР ФАЙЛА .env ДЛЯ НАСТРОЙКИ СИСТЕМЫ
=============================================================================

Скопируйте содержимое этого файла в .env и заполните необходимые значения.

Команда для копирования:
Windows (PowerShell): Copy-Item ENV_EXAMPLE.txt .env
Linux/Mac: cp ENV_EXAMPLE.txt .env

=============================================================================

# =============================================================================
# TELEGRAM BOT CONFIGURATION
# =============================================================================

# Telegram Bot Token (получить у @BotFather)
TELEGRAM_BOT_TOKEN=your_telegram_bot_token_here

# =============================================================================
# LLM / OPENAI CONFIGURATION
# =============================================================================

# OpenAI API Key (или OpenRouter API Key)
OPENAI_API_KEY=your_openai_api_key_here

# OpenAI Base URL (по умолчанию OpenRouter)
OPENAI_BASE_URL=https://openrouter.ai/api/v1

# Модель для использования
OPENAI_MODEL=openai/gpt-4

# =============================================================================
# BOT PERSONALITY
# =============================================================================

# Имя роли бота
BOT_ROLE_NAME=ИИ-ассистент

# Описание роли бота
BOT_ROLE_DESCRIPTION=Помогаю отвечать на вопросы

# Системный промпт (альтернатива SYSTEM_PROMPT_FILE)
# SYSTEM_PROMPT="Ты полезный ИИ-ассистент"

# Путь к файлу с системным промптом
SYSTEM_PROMPT_FILE=prompts/music_consultant.txt

# =============================================================================
# LOGGING CONFIGURATION
# =============================================================================

# Уровень логирования (DEBUG, INFO, WARNING, ERROR, CRITICAL)
LOG_LEVEL=INFO

# Путь к директории для логов
LOG_FILE_PATH=logs/

# =============================================================================
# CONTEXT CONFIGURATION
# =============================================================================

# Максимальное количество сообщений в контексте (0 = без ограничений)
MAX_CONTEXT_MESSAGES=20

# =============================================================================
# DATABASE CONFIGURATION
# =============================================================================

# URL базы данных (SQLite)
DATABASE_URL=sqlite+aiosqlite:///./data/telegram_bot.db

# =============================================================================
# API CONFIGURATION
# =============================================================================

# Порт для API сервера (на хосте)
API_PORT=8004

# =============================================================================
# FRONTEND CONFIGURATION
# =============================================================================

# Порт для Frontend сервера (на хосте)
FRONTEND_PORT=3004

# URL API для frontend (для production используйте IP сервера)
NEXT_PUBLIC_API_URL=http://89.223.67.136:8004

# =============================================================================
# SERVER CONFIGURATION
# =============================================================================

# IP адрес сервера для production
SERVER_IP=89.223.67.136

# =============================================================================
# DOCKER / DEPLOYMENT CONFIGURATION
# =============================================================================

# Версии образов для production
BOT_VERSION=latest
API_VERSION=latest
FRONTEND_VERSION=latest

# Build configuration
BUILD_DATE=
VCS_REF=

# =============================================================================
# ADVANCED CONFIGURATION (необязательно)
# =============================================================================

# Python unbuffered mode (для Docker)
# PYTHONUNBUFFERED=1

# Next.js telemetry
# NEXT_TELEMETRY_DISABLED=1

# Hot reload в development
# RELOAD=true
# WATCHFILES_FORCE_POLLING=true
# WATCHPACK_POLLING=true
